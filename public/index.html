document.addEventListener('DOMContentLoaded', () => { const chatMessages = document.getElementById('chat-messages'); const userInput = document.getElementById('user-input'); const sendButton = document.getElementById('send-button'); const typingIndicator = document.getElementById('typing-indicator'); // --- LOGIKA PASSWORD --- const overlay = document.getElementById('password-overlay'); const passInput = document.getElementById('app-pass'); const loginBtn = document.getElementById('login-btn'); const errorMsg = document.getElementById('pass-error'); loginBtn.onclick = () => { if (passInput.value === '19jutapekerjaan') { overlay.style.display = 'none'; userInput.focus(); } else { errorMsg.style.display = 'block'; passInput.value = ''; } }; // --- LOGIKA CHAT --- function appendMessage(role, text) { const messageDiv = document.createElement('div'); messageDiv.className = `message ${role}-message`; if (role === 'assistant') { // Mengubah markdown menjadi HTML (Kotak Kode) messageDiv.innerHTML = marked.parse(text); } else { const p = document.createElement('p'); p.textContent = text; messageDiv.appendChild(p); } chatMessages.appendChild(messageDiv); // Memicu Prism untuk mewarnai kode yang baru muncul if (window.Prism) { Prism.highlightAllUnder(messageDiv); } chatMessages.scrollTop = chatMessages.scrollHeight; } async function sendMessage() { const message = userInput.value.trim(); if (!message) return; appendMessage('user', message); userInput.value = ''; typingIndicator.classList.add('visible'); try { // Pastikan URL API Cloudflare Anda benar di sini const response = await fetch('/api/chat', { method: 'POST', body: JSON.stringify({ message }), headers: { 'Content-Type': 'application/json' } }); const data = await response.json(); typingIndicator.classList.remove('visible'); appendMessage('assistant', data.response); } catch (error) { typingIndicator.classList.remove('visible'); appendMessage('assistant', "Maaf, terjadi kesalahan teknis."); } } sendButton.onclick = sendMessage; userInput.onkeypress = (e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); } }; });
